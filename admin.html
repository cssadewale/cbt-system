⚙️ Setup & SQL Guide

Run these once in Supabase Dashboard → SQL Editor. Run them in order.

Step 1 — Enable RLS on Both Tables
ALTER TABLE exams ENABLE ROW LEVEL SECURITY; ALTER TABLE results ENABLE ROW LEVEL SECURITY;
Step 2 — Exams Isolation Policies
CREATE POLICY "Teachers select own exams" ON exams FOR SELECT USING (auth.uid() = teacher_id); CREATE POLICY "Teachers insert own exams" ON exams FOR INSERT WITH CHECK (auth.uid() = teacher_id); CREATE POLICY "Teachers update own exams" ON exams FOR UPDATE USING (auth.uid() = teacher_id); CREATE POLICY "Teachers delete own exams" ON exams FOR DELETE USING (auth.uid() = teacher_id);
Step 3 — Results Isolation Policies
CREATE POLICY "Teachers select own results" ON results FOR SELECT USING (auth.uid() = (SELECT teacher_id FROM exams WHERE id = exam_id)); CREATE POLICY "Students can submit results" ON results FOR INSERT WITH CHECK (true); CREATE POLICY "Teachers update own results" ON results FOR UPDATE USING (auth.uid() = (SELECT teacher_id FROM exams WHERE id = exam_id)); CREATE POLICY "Teachers delete own results" ON results FOR DELETE USING (auth.uid() = (SELECT teacher_id FROM exams WHERE id = exam_id));
Step 4 — Create Profiles Table (Teacher Management)

This table stores each teacher's status (pending/active/inactive) and role (teacher/admin). Required for approval workflow.

CREATE TABLE IF NOT EXISTS profiles ( id uuid PRIMARY KEY REFERENCES auth.users(id) ON DELETE CASCADE, email text NOT NULL, full_name text, role text NOT NULL DEFAULT 'teacher', status text NOT NULL DEFAULT 'pending', created_at timestamptz DEFAULT now(), updated_at timestamptz DEFAULT now() ); ALTER TABLE profiles ENABLE ROW LEVEL SECURITY; -- Teachers can read their own profile CREATE POLICY "Users read own profile" ON profiles FOR SELECT USING (auth.uid() = id); -- Allow insert on signup (via trigger below) CREATE POLICY "Allow profile insert" ON profiles FOR INSERT WITH CHECK (true); -- Trigger: auto-create profile on signup CREATE OR REPLACE FUNCTION handle_new_user() RETURNS trigger AS $$ BEGIN INSERT INTO profiles (id, email, full_name, role, status) VALUES ( NEW.id, NEW.email, NEW.raw_user_meta_data->>'display_name', 'teacher', 'pending' ) ON CONFLICT (id) DO NOTHING; RETURN NEW; END; $$ LANGUAGE plpgsql SECURITY DEFINER; DROP TRIGGER IF EXISTS on_auth_user_created ON auth.users; CREATE TRIGGER on_auth_user_created AFTER INSERT ON auth.users FOR EACH ROW EXECUTE FUNCTION handle_new_user();
Step 5 — Add Anti-Cheat Columns to Results
ALTER TABLE results ADD COLUMN IF NOT EXISTS answers_data JSONB; ALTER TABLE results ADD COLUMN IF NOT EXISTS violations integer DEFAULT 0; ALTER TABLE results ADD COLUMN IF NOT EXISTS violation_log JSONB;
Step 6 — Set Your Admin Email (IMPORTANT)

Edit the line below in teacher.html and admin.html to your actual admin email address. This identifies who is the platform admin.

// In both teacher.html and admin.html, find this line: const ADMIN_EMAIL = 'admin@cbtpro.com'; // Change it to your email: const ADMIN_EMAIL = 'yourname@yourdomain.com';

Then in Supabase, manually set your profile status to 'active' and role to 'admin':

-- Run this with YOUR actual user ID from Supabase Auth dashboard INSERT INTO profiles (id, email, role, status) VALUES ('YOUR-USER-UUID-HERE', 'admin@cbtpro.com', 'admin', 'active') ON CONFLICT (id) DO UPDATE SET role='admin', status='active';
